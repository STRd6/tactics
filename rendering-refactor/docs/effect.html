    module.exports = (I={}, self=Core(I)) ->

  perform: ({position, tileAt, characterAt, message}) ->
    if tile = tileAt(position)
      # Fireball Effect
      Object.extend tile,
        opaque: false
        solid: false
        features: []
        sprite: lavaSprites.rand()

      if character = characterAt(position)
        message("#{character.name()} is burned!")
        character.damage(2)
